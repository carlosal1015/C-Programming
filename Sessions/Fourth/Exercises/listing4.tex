\question Sospechas que hay algunos errores en el siguiente programa. ¿Puede encontrarlos?

\begin{minted}[linenos]{c}
int main(void)
{
	int=1,
	float n;
	printf("watch out! Here come a bunch of fractions!\n");
	while(i<30)
		n=1/i;
		printf(" %f", n);
		
printf("That's all, folks!\n");
return;
}
\end{minted}

\begin{solution}
\begin{enumerate}[label={Línea \arabic*:},leftmargin=1.2cm]
	\item Debe ir la cabecera \mintinline{c}{#include <stdio.h>}.\setcounter{enumi}{2}

	\item Debe acabar en punto y coma, no en coma.
	
	\item Debe ir una coma, no un punto y coma, entre g y h.
	
	\item Debe haber al menos un dígito antes de la \mintinline{c}{e}. Ya sea \mintinline{c}{1e21} o \mintinline{c}{1.0e21} está bien, aunque es bastante grande.
	
	\item Bien, al menos en términos de sintaxis.
	
	\item Use \}, no ).
\end{enumerate}

Aquí hay una posible versión correcta:

	\caption{Programa \texttt{exercise4\_2.c}.}
	\label{lst:4.2}
\end{listing}
\end{solution}
